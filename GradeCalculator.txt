using System;
namespace std
{
    class Program
    {
        public static void Main(string[] args)
        {
            //Welome Message
            Console.WriteLine("Welcome to Grade Calculator. ");
            Console.Write("Enter Number of Subjects: ");
            int subjects=Convert.ToInt32(Console.ReadLine());
            double totalMarks=0;
            //Condition
            for(int i=1;i<=subjects;i++)
            {
                Console.Write($"Enter Marks out of 100 for Subject {i} : ");
                double marks=double.Parse(Console.ReadLine());
                if(marks>=0&&marks<=100)
                {
                    totalMarks+=marks;
                }
                else
                {
                    Console.WriteLine("Invalid marks entered. Marks should be between 0 and 100.");
                    i--; // Decrement i to re-enter marks for the same subject.
                }
                double averageMarks=totalMarks/subjects;
                string grade;
                    if (averageMarks >= 85)
            {
                grade = "A";
            }
            else if (averageMarks >= 80)
            {
                grade = "B";
            }
            else if (averageMarks >= 70)
            {
                grade = "C";
            }
            else if (averageMarks >= 60)
            {
                grade = "D";
            }
            else if (averageMarks >= 50)
            {
                grade = "E";
            }
            else
            {
                grade = "F";
            }
            Console.WriteLine($"Average marks: {averageMarks:F2}");
            Console.WriteLine($"Grade: {grade}");
            }
        }
    }
}
